/// Holds information about Iris flower to be classified.
[<CLIMutable>]
type IrisData = {
    [<LoadColumn(0)>]
    Label : float32

    [<LoadColumn(1)>]
    SepalLength : float32
        
    [<LoadColumn(2)>]
    SepalWidth : float32
        
    [<LoadColumn(3)>]
    PetalLength : float32
        
    [<LoadColumn(4)>]
    PetalWidth : float32
} 

/// Result of Iris classification. The array holds probability of the flower to be one of setosa, virginica or versicolor.
